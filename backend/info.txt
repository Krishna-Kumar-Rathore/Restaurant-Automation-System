const fetchFoodItems = async () => {
    try {
        const db = mongoose.connection.useDb("GoFoodItems");  // Specify GoFoodItem database
        const fetched_data = db.collection("food_items");
        
        // Fetch all documents in food_items collection
        const data = await fetched_data.find({}).toArray();

        if (data.length === 0) {
            console.log("The collection 'food_items' is empty.");
        } else {
            console.log("Fetched Data:", data);
        }
    } catch (error) {
        console.error("Error fetching data:", error);
    }
};

// Connect to MongoDB and then fetch data
connectToMongo().then(() => fetchFoodItems());

import React, { useEffect, useState } from 'react';
import Navbar from '../Components/Navbar';
import Footer from '../Components/Footer';
import Card from '../Components/Card';
import Carousal from '../Components/Carousal';

function Home() {
  const [foodCat, setFoodCat] = useState([]); // Corrected to useState
  const [foodItem, setFoodItem] = useState([]); // Corrected to useState

  const loadData = async () => {
    try {
      let response = await fetch("localhost:5000/api/foodData", { // Added protocol  http://localhost:5000/api/foodData
        method: "POST",
        headers: {
          'Content-Type': 'application/json'
        }
      });
      response = await response.json();
      setFoodItem(response[0]);
      setFoodCat(response[1]);
    } catch (error) {
      console.error("Error fetching data:", error);
    }
  };

  useEffect(() => {
    loadData();
  }, []);

  return (
    <div>
      <div><Navbar /></div>
      <div><Carousal /></div>
      <div className='container'>
        {
          foodCat.length > 0 // Corrected comparison to check length
            ? foodCat.map((data) => {
              return <div>
              <div key={data._id} className='fs-3 m-3' > {/* Added key for map */}
              {data.CategoryName}
            </div>
              <hr />
              { foodItem.length > 0 //foodItem !== []
              ? foodItem.filter((item) => item.CategoryName === data.CategoryName).map(filterItems=>{
                return (
                  <div  key={filterItems._id} >
                    <Card/>
                  </div>
                )
              })
               : <div>No such Data Found</div> } 

            </div>
            }   
            )
            : ".........."
        }
        <Card />
      </div>
      <div style={{ color: 'whitesmoke' }}><Footer /></div>
    </div>
  );
}

export default Home;
